-- Создание таблицы сотрудников
CREATE TABLE Сотрудники (
    id SERIAL PRIMARY KEY,
    имя VARCHAR(50),
    должность VARCHAR(100),
    дата_приема DATE,
    активен BOOLEAN
);

-- Создание таблицы отделов
CREATE TABLE Отделы (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    руководитель_id INTEGER REFERENCES Сотрудники(id)
);

-- Создание таблицы проектов
CREATE TABLE Проекты (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    описание TEXT,
    дата_начала DATE,
    дата_окончания DATE,
    статус BOOLEAN
);

-- Создание таблицы задач
CREATE TABLE Задачи (
    id SERIAL PRIMARY KEY,
    проект_id INTEGER REFERENCES Проекты(id),
    название VARCHAR(100),
    описание TEXT,
    дата_назначения DATE,
    статус BOOLEAN
);
