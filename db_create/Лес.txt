-- Создание таблицы деревьев
CREATE TABLE Деревья (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    тип VARCHAR(50),
    высота INTEGER,
    диаметр INTEGER,
    возраст INTEGER,
    местоположение VARCHAR(100)
);

-- Создание таблицы видов деревьев
CREATE TABLE Виды_деревьев (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    описание TEXT
);

-- Создание таблицы лесов
CREATE TABLE Леса (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    площадь INTEGER,
    местоположение VARCHAR(100)
);
