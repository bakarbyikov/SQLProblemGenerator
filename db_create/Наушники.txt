-- Создание таблицы брендов
CREATE TABLE Бренды (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    страна VARCHAR(50),
    год_основания INTEGER
);

-- Создание таблицы типов наушников
CREATE TABLE Типы (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    описание TEXT
);

-- Создание таблицы наушников
CREATE TABLE Наушники (
    id SERIAL PRIMARY KEY,
    название VARCHAR(100),
    бренд_id INTEGER REFERENCES Бренды(id),
    тип_id INTEGER REFERENCES Типы(id),
    цвет VARCHAR(50),
    цена INTEGER,
    беспроводные BOOLEAN
);
